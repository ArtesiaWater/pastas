            line = f.readline()
            header = dict()
            while line not in ['\n', '']:
                propval = line.split(',')
                prop = propval[0];
                prop = prop.replace(':', '')
                prop = prop.strip()
                val = propval[1];
                if propval[2] != '':
                    val = val + ' ' + propval[2].replace(':', '') + ' ' + propval[3]
                header[prop] = val
                line = f.readline()

            # lees gat
            while line == '\n':
                line = f.readline()

            # lees referentieniveaus
            ref = dict()
            while line not in ['\n', '']:
                propval = line.split(',')
                prop = propval[0];
                prop = prop.replace(':', '')
                prop = prop.strip()
                if len(propval) > 1:
                    val = propval[1];
                    ref[prop] = val
                line = f.readline()

            # lees gat
            while line == '\n':
                line = f.readline()

            # lees meta-informatie
            metaList = list()
            line = line.strip();
            properties = line.split(',')
            line = f.readline()
            while line not in ['\n', '']:
                meta = dict()
                line = line.strip();
                values = line.split(',')
                for i in range(0, len(values)):
                    meta[properties[i]] = values[i];
                metaList.append(meta)
                line = f.readline()

            # lees gat
            while line == '\n':
                line = f.readline()

            line = line.strip();
            titel = line.split(',')

                titel.remove('')
